So funktioniert die Google Sheets Verbindung

Schritt 1 – Öffne sheets.google.com und erstelle eine neue leere Tabelle.

Schritt 2 – Klick oben auf Erweiterungen → Apps Script.

Schritt 3 – Lösch den vorhandenen Code, füge das Script unten ein und speichere (Strg+S).

Schritt 4 – Klick auf Deployen → Neue Deployment. Typ: “Web-App”. Zugriff: “Jeder”. Dann deployen und die URL kopieren.

Schritt 5 – Füge die URL unten ein und speichere. Ab jetzt wird jedes Workout automatisch eingetragen.

function doPost(e) {
  try {
    var data = JSON.parse(e.postData.contents);
    var ss = SpreadsheetApp.getActiveSpreadsheet();
    var sheet = ss.getSheetByName("Trainings")
                || ss.insertSheet("Trainings");
    if (sheet.getLastRow() === 0) {
      sheet.appendRow(["Datum","Tag","Uebung",
        "Sets","Reps","Gewicht (kg)","Woche"]);
    }
    data.exercises.forEach(function(ex) {
      sheet.appendRow([data.date, data.day,
        ex.name, ex.sets, ex.reps,
        ex.weight || "Bodyweight", data.week]);
    });
    return ContentService.createTextOutput(
      JSON.stringify({status:"ok"}))
      .setMimeType(ContentService.MimeType.JSON);
  } catch(err) {
    return ContentService.createTextOutput(
      JSON.stringify({status:"error",msg:err.message}))
      .setMimeType(ContentService.MimeType.JSON);
  }
}
